NodeJsPractice/
├── public/                        # Каталог для статических файлов, таких как CSS и JavaScript.
│   ├── css/
│   │   └── styles.css             # Файл стилей для веб-страницы.
│   ├── js/
│   │   └── script.js              # Файл JavaScript для клиентской логики.
│   └── index.html                 # Главная HTML-страница.
├── routes/
│   └── users.js                   # Файл маршрутов для обработки запросов, связанных с пользователями.
├── views/
│   └── users.ejs                  # Шаблон для отображения списка пользователей.
├── models/
│   └── User.js                    # Модель пользователя и функции для работы с базой данных PostgreSQL.
├── node_modules/                  # Каталог, где npm хранит установленные модули.
├── app.js                         # Главный файл приложения, который запускает сервер и настраивает маршруты.
├── package.json                   # Файл конфигурации проекта, содержащий информацию о зависимостях и скриптах.
└── package-lock.json              # Файл, автоматически генерируемый npm, фиксирующий версии установленных пакетов.
